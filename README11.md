use Illuminate\Support\Facades\DB; // クェリビルダ

public functio index()
{
  $events = DB::table('events')
    ->orderBy('start_date', 'asc') // 開始日時順
    ->paginate(10); // 10件ずつ

  return view('manager.events.index', compact('events')); // 変数をviewに渡す
}
